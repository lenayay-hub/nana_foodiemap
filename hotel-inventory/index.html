            <h2 class="text-xl font-bold mb-4 text-center">簽名確認</h2>

    <script>

        const API_URL = "https://script.google.com/macros/s/AKfycbwJv-qdQcoqowK1JYqSttVGYiSh0nW-SO1-6N_D9pSHok36i7XMt_uL_oRMKJGxFWw/exec";

        

        async function init() {

            try {

                document.getElementById('datePicker').valueAsDate = new Date();

                const res = await fetch(API_URL + "?action=getData");

                const allItems = await res.json();

                

                if(allItems.error) throw new Error(allItems.error);


                const hotels = [...new Set(allItems.map(d => d.hotel))];

                document.getElementById('hotelSelect').innerHTML = hotels.map(h => `<option value="${h}">${h}</option>`).join('');

                

                allItems.forEach(d => createItemRow(d.hotel, d.item));

                

                document.getElementById('loading').classList.add('hidden');

                document.getElementById('content').classList.remove('hidden');

                filterList();

            } catch (err) {

                document.getElementById('loading').innerText = "連線失敗: " + err.message;

            }

        }


        function createItemRow(hotel, item) {

            const list = document.getElementById('itemsList');

            const div = document.createElement('div');

            div.className = 'item-row';

            div.dataset.hotel = hotel;

            div.dataset.item = item.toLowerCase();

            div.innerHTML = `

                <div class="flex justify-between items-center mb-1">

                    <span class="text-2xl font-black text-gray-900">${item}</span>

                    <input type="text" readonly onclick="openNumPad(this)" class="w-32 p-2 text-center text-3xl font-bold border-b-2 border-blue-500" placeholder="0">

                </div>

                <div class="flex gap-2 items-center">

                    <input type="text" class="flex-1 p-1.5 bg-gray-50 rounded text-sm note-input" placeholder="備註...">

                    <button onclick="openSign(this.parentElement.parentElement)" class="px-4 py-1.5 bg-blue-600 text-white rounded font-bold text-sm">簽名</button>

                </div>

            `;

            list.appendChild(div);

        }


        function filterList() {

            const hotel = document.getElementById('hotelSelect').value;

            const search = document.getElementById('search').value.toLowerCase();

            document.querySelectorAll('.item-row').forEach(row => {

                row.style.display = (row.dataset.hotel === hotel && row.dataset.item.includes(search)) ? 'block' : 'none';

            });

        }

        

        // 簽名板與鍵盤邏輯保持不變...

        let targetRow = null, targetInput = null;

        const canvas = document.getElementById('signPad');

        const ctx = canvas.getContext('2d');

        let isDrawing = false;

        function openNumPad(inputEl) { targetInput = inputEl; document.getElementById('numDisplay').innerText = inputEl.value || "0"; document.getElementById('numPadModal').classList.remove('hidden'); }

        function handleNum(n) { let d = document.getElementById('numDisplay'); if (n === 'del') d.innerText = d.innerText.length > 1 ? d.innerText.slice(0, -1) : "0"; else d.innerText = (d.innerText === "0" ? "" : d.innerText) + n; }

        function closeNumPad() { if (targetInput) targetInput.value = document.getElementById('numDisplay').innerText; document.getElementById('numPadModal').classList.add('hidden'); }

        function openSign(rowEl) { targetRow = rowEl; document.getElementById('signModal').classList.remove('hidden'); ctx.clearRect(0,0, canvas.width, canvas.height); }

        function closeSign() { document.getElementById('signModal').classList.add('hidden'); }

        canvas.onmousedown = () => isDrawing = true; canvas.onmousemove = (e) => { if(isDrawing) { ctx.lineTo(e.offsetX, e.offsetY); ctx.stroke(); } }; canvas.onmouseup = () => isDrawing = false;

        canvas.ontouchstart = (e) => { isDrawing = true; ctx.beginPath(); ctx.moveTo(e.touches[0].clientX - canvas.getBoundingClientRect().left, e.touches[0].clientY - canvas.getBoundingClientRect().top); };

        canvas.ontouchmove = (e) => { if(isDrawing) { ctx.lineTo(e.touches[0].clientX - canvas.getBoundingClientRect().left, e.touches[0].clientY - canvas.getBoundingClientRect().top); ctx.stroke(); } };


        async function saveSign() {

            const payload = {

                hotel: document.getElementById('hotelSelect').value,

                date: document.getElementById('datePicker').value,

                item: targetRow.querySelector('span').innerText,

                count: targetRow.querySelector('input').value,

                note: targetRow.querySelector('.note-input').value,

                timestamp: new Date().toLocaleString()

            };

            await fetch(API_URL, { method: 'POST', mode: 'no-cors', body: JSON.stringify(payload) });

            alert("紀錄已成功送出！");

            closeSign();

        }

        init();
